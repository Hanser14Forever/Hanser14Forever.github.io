class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        stack<int> s1, s2;
        while (l1) {
            s1.push(l1->val);
            l1 = l1->next;
        }
        while (l2) {
            s2.push(l2->val);
            l2 = l2->next;
        }

        int carry = 0;
        ListNode* head = nullptr;
        while (!s1.empty() || !s2.empty() || carry) {
            int s = (!s1.empty() ? s1.top() : 0) + (!s2.empty() ? s2.top() : 0) + carry;
            if (!s1.empty()) s1.pop();
            if (!s2.empty()) s2.pop();
            carry = s / 10;
            ListNode* curnode = new ListNode(s%10);
            curnode->next = head;
            head = curnode;
        }
        return head;
    }
};
